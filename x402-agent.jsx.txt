// ─── UTILS ────────────────────────────────────────────────────────────────────
const sleep = (ms) => new Promise((r) => setTimeout(r, ms));
const fmtAmt = (n) => `$${Number(n).toFixed(4)}`;
const fmtHash = (h) => h ? `${h.slice(0, 8)}…${h.slice(-6)}` : "—";
const fmtTime = (ms) => `${ms}ms`;
const uid = () => Math.random().toString(36).slice(2, 10).toUpperCase();

const SECURITY_CHECKS = [
  { id: "sig",   label: "Signature verification",    icon: "🔐" },
  { id: "replay",label: "Replay attack prevention",  icon: "🛡️" },
  { id: "amt",   label: "Amount bounds check",       icon: "⚖️" },

// ─── SIMULATED x402 AGENT ENGINE ─────────────────────────────────────────────
async function runPaymentFlow(config, onStep) {
  const startMs = Date.now();
  const txId = uid();

  const step = async (id, label, detail, state = "running") => {
    onStep({ id, label, detail, state, ts: Date.now() - startMs });
    await sleep(340 + Math.random() * 200);
  };

  // 1 — Initial request
  await step("req", "HTTP GET →", config.endpoint, "running");
  await step("req", "HTTP GET →", config.endpoint, "done");

  // 2 — 402 received
  await step("recv402", "← 402 Payment Required", `amount=${fmtAmt(config.amount)} network=${NETWORKS[config.network]?.name}`, "running");
  await step("recv402", "← 402 Payment Required", `amount=${fmtAmt(config.amount)} network=${NETWORKS[config.network]?.name}`, "done");

  // 3 — Security checks
  for (const chk of SECURITY_CHECKS) {
    onStep({ id: `sec_${chk.id}`, label: `${chk.icon} ${chk.label}`, detail: "checking…", state: "running", ts: Date.now() - startMs });
    await sleep(160 + Math.random() * 120);
    onStep({ id: `sec_${chk.id}`, label: `${chk.icon} ${chk.label}`, detail: "PASSED", state: "done", ts: Date.now() - startMs });
  }

  // 4 — Build payment payload
  await step("build", "🛠  Building signed payload", "EIP-3009 transferWithAuthorization", "running");
  await step("build", "🛠  Building signed payload", "EIP-3009 transferWithAuthorization", "done");

  // 5 — Submit to facilitator
  await step("submit", "📡 Submitting to facilitator", "POST /verify → CDP", "running");
  await sleep(500 + Math.random() * 300);

  // 6 — Simulate rare failure → auto-retry
  const shouldRetry = config.retryDemo;
  if (shouldRetry) {
    onStep({ id: "submit", label: "📡 Submitting to facilitator", detail: "⚠️ timeout — retrying (1/3)…", state: "warn", ts: Date.now() - startMs });
    await sleep(600);
    onStep({ id: "submit", label: "📡 Submitting to facilitator", detail: "⚠️ timeout — retrying (2/3)…", state: "warn", ts: Date.now() - startMs });
    await sleep(400);
  }
  onStep({ id: "submit", label: "📡 Submitting to facilitator", detail: "settlement confirmed ✓", state: "done", ts: Date.now() - startMs });

  // 7 — On-chain settlement
  const fakeHash = "0x" + Array.from({ length: 64 }, () => Math.floor(Math.random() * 16).toString(16)).join("");
  await step("settle", "⛓  On-chain settlement", `tx: ${fakeHash.slice(0, 18)}…`, "running");
  await sleep(400);
  onStep({ id: "settle", label: "⛓  On-chain settlement", detail: `tx: ${fakeHash.slice(0, 18)}… FINALIZED`, state: "done", ts: Date.now() - startMs });

  // 8 — Resource delivered
  await step("deliver", "✅ Resource delivered", "200 OK — access granted", "done");

  const elapsed = Date.now() - startMs;
  return { txId, hash: fakeHash, elapsed, amount: config.amount, network: config.network };
}

// ─── SUB-COMPONENTS ──────────────────────────────────────────────────────────
function StepRow({ step }) {
  const clr = step.state === "done" ? C.green : step.state === "warn" ? C.yellow : step.state === "running" ? C.accent : C.muted;
  const icon = step.state === "done" ? "✓" : step.state === "warn" ? "⚠" : step.state === "running" ? "⟳" : "·";
  return (
    <div style={{ display: "flex", alignItems: "flex-start", gap: 10, padding: "5px 0", borderBottom: `1px solid ${C.border}22` }}>
      <span style={{ color: clr, fontSize: 13, width: 14, flexShrink: 0, marginTop: 1, animation: step.state === "running" ? "spin 1s linear infinite" : "none" }}>{icon}</span>
      <div style={{ flex: 1, minWidth: 0 }}>
        <span style={{ color: step.state === "done" ? C.textHi : C.text, fontSize: 13, fontFamily: "monospace" }}>{step.label}</span>
        <span style={{ color: C.muted, fontSize: 12, fontFamily: "monospace", marginLeft: 8 }}>{step.detail}</span>
      </div>
      <span style={{ color: C.muted, fontSize: 11, fontFamily: "monospace", flexShrink: 0 }}>{fmtTime(step.ts)}</span>
    </div>
  );
}

function StatBox({ label, value, accent }) {
  return (
    <div style={{ background: C.panel, border: `1px solid ${C.border}`, borderRadius: 8, padding: "12px 16px", flex: 1, minWidth: 120 }}>
      <div style={{ color: C.muted, fontSize: 11, textTransform: "uppercase", letterSpacing: 1 }}>{label}</div>
      <div style={{ color: accent || C.textHi, fontSize: 20, fontWeight: 700, fontFamily: "monospace", marginTop: 4 }}>{value}</div>
    </div>
  );
}

function Badge({ children, color }) {
  return (
    <span style={{ background: color + "22", color: color, border: `1px solid ${color}44`, borderRadius: 4, fontSize: 11, padding: "2px 7px", fontFamily: "monospace" }}>
      {children}
    </span>
  );
}
